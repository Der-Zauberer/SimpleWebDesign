    swd.configureLanguages({
        languages: [
            { locale: 'en', src: 'i18n/en.properties' },
            { locale: 'de', src: 'i18n/de.properties' },
            { locale: 'es', src: 'i18n/es.properties' }
        ],
        fallback: 'en'
    })
    swd.setLanguage(navigator.language); //Load browser language or use fallback
</script>
